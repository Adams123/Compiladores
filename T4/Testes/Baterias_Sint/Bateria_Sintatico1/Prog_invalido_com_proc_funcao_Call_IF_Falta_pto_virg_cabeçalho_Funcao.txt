PROGRAM TESTE2;
{teste de comentário em programa}
VARIABLES 
	INTEGER a;
	REAL b, N1, N2, N3, MediaAlunos;
	
PROCEDURE FindFormula; 
PARAMETERS
INTEGER x;
REAL y;
	BEGIN
		WHILE X ! 0 DO
		IF X<0 THEN SET y = 10 - 4.5 * x
		ELSE SET y = 4.5 * -x + 10
		ENDIF;
		WRITE y;
		READ x
		ENDWHILE
	END;
	
FUNCTION media RETURNS REAL
PARAMETERS
INTEGER a;
INTEGER b;
INTEGER c;
VARIABLES 
REAL SUM, AVERAGE;

BEGIN
SET Sum = a+b+c;
SET AVERAGE=SUM/3;
RETURN (AVERAGE)
END;

{main}		 

BEGIN
	READ a;
	CALL FindFormula(a,b);
	WRITE b;
	READ N1;READ N2;READ N3;
	SET MediaAlunos = media(N1,N2,N3)
END.
