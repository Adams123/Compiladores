procedure findformula; 
parameters
integer x;
real y;
	begin
		while x ! 0 do
		if x<0 then set y = 10 - 4.5 * x
		else set y = 4.5 * -x + 10
		endif;
		write y;
		read x
		endwhile

parameters
integer a;

return (average)
end;
